# [storybook-addon-shiki](https://github.com/lukethacoder/storybook-addon-shiki)

[Shiki Documentation](https://shiki.style/) for more configuration information.

Quick example usage of code snippets with this addon enabled.

## Language Support

Support for any language [shiki](https://shiki.style/languages) supports

### HTML

<Source language="html" code={`
  <html lang="en">
    <head>
      <title>hello world</title>
    </head>
    <body>
      <!-- comment block -->
      <main>
        <h1 class="text-2xl md:text-5xl">hello world
      </main>  
    </body>
  </html>
`} />

### TSX

<Source language="tsx" code={`
  /**
  * Comment block
  */
  import React from 'react';
  import type { ShikiAddonOptions } from '../types';
  
  interface Props {
    children?: React.ReactNode;
    [key: string]: unknown;
  }
  
  /**
  * Component def
  */
  export function SyntaxHighlighter({
    children,
    language = 'jsx',
    showLineNumbers = false,
    ...rest
  }: Props) {
    const [options, setOptions] = React.useState<ShikiAddonOptions>({});
  
    React.useEffect(() => {
      optionsPromise.then(setOptions);
    }, []);
  
    const code = typeof children === 'string' ? children : String(children ?? '');
  
    return (
      <ShikiHighlighter language={language} showLineNumbers={showLineNumbers} options={options} {...rest}>
        {code}
      </ShikiHighlighter>
    );
  }
`} />


### Svelte

<Source language="svelte" code={`
  <script lang="ts">
    const greet = (name: string) => console.log(\`Hello ${name}, welcome to Svelte!\`);
  </script>
  
  <button onclick={greet}>click me</button>
  
  <style>
  	button {
  		font-size: 2em;
  	}
  </style>
`} />
